10def read_vertices(filename):
11 """读取多边形顶点数据"""
12 vertices = []
13 with open(filename, 'r') as f:
14 for line in f:
15 x, y = map(float, line.strip().split())
16 vertices.append([x, y])
17 return np.array(vertices)
18
19def read_grid_points(data_dir):
20 """读取网格点数据：固定读取 data/grid_X.txt 和 data/grid_Y.txt 文件"""
21 x_file = os.path.join(data_dir, 'grid_X.txt')
22 y_file = os.path.join(data_dir, 'grid_Y.txt')
23 with open(x_file, 'r') as f:
24 x_coords = list(map(float, f.read().split()))
25 with open(y_file, 'r') as f:
26 y_coords = list(map(float, f.read().split()))
27 return np.column_stack((x_coords, y_coords))

29def plot_polygon(vertices, grid_points, output_file=None, prefix=None):
30 """绘制多边形和网格点"""
31 fig, ax = plt.subplots(figsize=(8, 6))
32
33 # 绘制多边形（使用深青色线条和半透明填充）
34 ax.fill(vertices[:, 0], vertices[:, 1],
35 color='#2b8cbe', alpha=0.3)
36 ax.plot(vertices[:, 0], vertices[:, 1],
37 color='#045a8d', linewidth=2, linestyle='-')
38 ax.plot([vertices[-1, 0], vertices[0, 0]],
39 [vertices[-1, 1], vertices[0, 1]],
40 color='#045a8d', linewidth=2) # 闭合多边形
41
42 # 标记顶点（使用橙色圆点）
43 ax.scatter(vertices[:, 0], vertices[:, 1],
44 color='#fd8d3c', s=50, edgecolor='white',
45 linewidth=1, zorder=3)
46
47 # 绘制网格点
48 ax.scatter(grid_points[:, 0], grid_points[:, 1],
49 color='#7f7f7f', s=10, alpha=0.5,
50 marker='.', label='Grid Points')
51
52 # 设置图形属性
53 ax.set_title('Polygon Region with ' + (grid_points.shape[0]).__str__() + ' Grid Points', fontsize=14, pad=20)
54 ax.set_xlabel('x', fontsize=12, labelpad=10)
55 ax.set_ylabel('y', fontsize=12, labelpad=10)
56 ax.grid(True, linestyle=':', color='gray', alpha=0.4)
57 ax.set_aspect('equal')
58
59 # 自动保存到项目数据文件夹
60 if prefix:
61 auto_save_path = os.path.join(data_dir, f'{prefix}_grid.png')
62 plt.savefig(auto_save_path, dpi=300, bbox_inches='tight', facecolor='white')
63 print(f"已自动保存到: {auto_save_path}")
64
65 if output_file:
66 plt.savefig(output_file, dpi=300, bbox_inches='tight', facecolor='white')
67 print(f"已保存到: {output_file}")
68
69 plt.show()
