10def segments_intersect(a, b, c, d, tol=1e-10):
11 """检查线段ab和cd是否相交，考虑数值误差"""
12 def ccw(A, B, C):
13 return (C[1]-A[1]) * (B[0]-A[0]) - (B[1]-A[1]) * (C[0]-A[0])
14
15 def point_on_segment(p, a, b):
16 """检查点p是否在线段ab上（考虑容忍度）"""
17 # 检查点p是否在a和b的边界框内
18 if (min(a[0], b[0]) - tol <= p[0] <= max(a[0], b[0]) + tol and
19 min(a[1], b[1]) - tol <= p[1] <= max(a[1], b[1]) + tol):
20 # 检查点p是否在直线ab上
21 cross = ccw(a, b, p)
22 return abs(cross) <= tol
23 return False
24
25 # 检查端点是否在另一条线段上
26 if point_on_segment(a, c, d) or point_on_segment(b, c, d) or \
27 point_on_segment(c, a, b) or point_on_segment(d, a, b):
28 return True
29
30 # 使用改进的跨立试验，考虑数值误差
31 o1 = ccw(a, c, d)
32 o2 = ccw(b, c, d)
33 o3 = ccw(c, a, b)
34 o4 = ccw(d, a, b)
35
36 # 如果两条线段严格相交
37 if (o1 * o2 < -tol) and (o3 * o4 < -tol):
38 return True
39
40 # 如果两条线段几乎共线，检查它们是否重叠
41 if abs(o1) <= tol and abs(o2) <= tol and abs(o3) <= tol and abs(o4) <= tol:
42 # 检查线段是否在x轴或y轴上有重叠
43 if (max(a[0], b[0]) >= min(c[0], d[0]) - tol and
44 min(a[0], b[0]) <= max(c[0], d[0]) + tol and
45 max(a[1], b[1]) >= min(c[1], d[1]) - tol and
46 min(a[1], b[1]) <= max(c[1], d[1]) + tol):
47 return True
48
49 return False

51def resolve_file(arg: str, data_dir: Path) -> Path:
52 """仅在 data_dir 中按文件名查找文件。"""
53 if '/' in arg or '\\' in arg or Path(arg).is_absolute():
54 raise ValueError(f"参数必须是文件名而不是路径：{arg}")
55
56 if not data_dir.exists() or not data_dir.is_dir():
57 raise FileNotFoundError(f"数据目录不存在: {data_dir}")
58
59 # 先尝试完全匹配文件名（包括扩展名）
60 for f in data_dir.iterdir():
61 if f.is_file() and f.name == arg:
62 return f
63
64 # 再尝试按 stem 匹配（去掉扩展名的名称）
65 for f in data_dir.iterdir():
66 if f.is_file() and f.stem == arg:
67 return f
68
69 raise FileNotFoundError(f"在 {data_dir} 中找不到名为 '{arg}' 的文件")

71def read_polygon_vertices(data_dir):
72 """读取多边形顶点"""
73 vertices_file = data_dir / "polygon_vertices.txt"
74 vertices = []
75 with open(vertices_file, 'r') as f:
76 for line in f:
77 x, y = map(float, line.strip().split())
78 vertices.append([x, y])
79 return np.array(vertices)
