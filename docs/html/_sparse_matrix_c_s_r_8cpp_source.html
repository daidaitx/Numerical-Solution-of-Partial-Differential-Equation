28// 构造函数：从稠密矩阵（二维 vector）构造 CSR 稀疏矩阵
29SparseMatrixCSR::SparseMatrixCSR(const vector<vector<double>>& dense) {
30 // 1. 检查矩阵是否为空
31 if (dense.empty() || dense[0].empty()) {
32 throw invalid_argument("矩阵行数和列数必须均大于0。");
33 }
34
35 // 2. 获取行数和列数
36 rows = dense.size();
37 cols = dense[0].size();
38
39 // 3. 检查是否是“矩形”矩阵（所有行的列数相同）
40 for (const auto& row : dense) {
41 if (row.size() != cols) {
42 throw invalid_argument("输入的稠密矩阵必须是矩形矩阵，每一行的列数必须相同。");
43 }
44 }
45
46 // 4. 初始化 CSR 数据结构
47 values.clear();
48 col_indices.clear();
49 row_ptrs.clear();
50
51 row_ptrs.resize(rows + 1, 0); // row_ptrs[i] 表示第 i 行在 values 中的起始位置
52
53 // 5. 遍历每一行、每一列，收集非零元素
54 for (size_t i = 0; i < rows; ++i) {
55 for (size_t j = 0; j < cols; ++j) {
56 double val = dense[i][j];
57 if (val != 0.0) {
58 values.push_back(val);
59 col_indices.push_back(j);
60 }
61 }
62 row_ptrs[i + 1] = values.size(); // 下一行的起始位置 = 当前 values 的大小
63 }
64}

66// 构造函数：从 COO 格式(1-based)的矩阵构造 CSR 稀疏矩阵
67SparseMatrixCSR::SparseMatrixCSR(
68 const size_t rows, const size_t cols,
69 const vector<size_t>& coo_row_indices,
70 const vector<size_t>& coo_col_indices,
71 const vector<double>& coo_values)
72 : rows(rows), cols(cols)
73{
74 // ===========================
75 // 1. 输入合法性检查
76 // ===========================
77 if (rows == 0 || cols == 0) {
78 throw invalid_argument("错误：矩阵的行数和列数必须均大于 0。");
79 }
80
81 const size_t nnz = coo_values.size();
82 if (coo_row_indices.size() != nnz || coo_col_indices.size() != nnz) {
83 throw invalid_argument("错误：COO 格式错误，行、列、值的数量必须一致。");
84 }
85
86 for (size_t i = 0; i < nnz; ++i) {
87 if (coo_row_indices[i] > rows || coo_col_indices[i] > cols) {
88 throw out_of_range("错误：COO 格式错误，行或列索引超出矩阵维度。");
89 }
90 if (coo_row_indices[i] == 0 || coo_col_indices[i] == 0) {
91 throw out_of_range("错误：COO 格式错误，行和列索引须为 1-based 索引。");
92 }
93 }
94
95 // 拷贝一份 0-based 索引的
96 vector<size_t> coo_row_indices_0b = coo_row_indices;
97 vector<size_t> coo_col_indices_0b = coo_col_indices;
98 for (size_t i = 0; i < nnz; ++i) {
99 --coo_row_indices_0b[i];
100 --coo_col_indices_0b[i];
101 }
102
103 // ===========================
104 // 2. 构造 COO 三元组 (row, col, value) 并排序（先行后列）
105 // ===========================
106 vector<tuple<size_t, size_t, double>> triples(nnz);
107 for (size_t i = 0; i < nnz; ++i) {
108 triples[i] = make_tuple(coo_row_indices_0b[i], coo_col_indices_0b[i], coo_values[i]);
109 }
110
111 // 按行优先，然后按列排序 => 保证每行内部 col 有序
112 sort(triples.begin(), triples.end(),
113 [](const tuple<size_t, size_t, double>& a,
114 const tuple<size_t, size_t, double>& b) {
115 if (get<0>(a) != get<0>(b))
116 return get<0>(a) < get<0>(b); // 按行号升序
117 return get<1>(a) < get<1>(b); // 同一行，按列号升序
118 });
119
120 // ===========================
121 // 3. 统计每行非零元个数，构建 row_ptrs（前缀和）
122 // ===========================
123 vector<size_t> row_nnz_counts(rows, 0);
124 for (const auto& t : triples) {
125 size_t row = get<0>(t);
126 row_nnz_counts[row]++;
127 }
128
129 row_ptrs.resize(rows + 1, 0);
130 for (size_t r = 0; r < rows; ++r) {
131 row_ptrs[r + 1] = row_ptrs[r] + row_nnz_counts[r];
132 }
133
134 // ===========================
135 // 4. 填充 values 和 col_indices
136 // ===========================
137 const size_t nnz_sorted = triples.size(); // == nnz
138 this->values.resize(nnz_sorted);
139 this->col_indices.resize(nnz_sorted);
140
141 // 每行当前写入的位置
142 vector<size_t> row_write_positions(rows, 0);
143
144 for (size_t i = 0; i < nnz_sorted; ++i) {
145 size_t row = get<0>(triples[i]);
146 size_t col = get<1>(triples[i]);
147 double value = get<2>(triples[i]);
148
149 size_t start = row_ptrs[row]; // 该行的起始索引
150 size_t pos = start + row_write_positions[row];
151
152 this->values[pos] = value;
153 this->col_indices[pos] = col;
154
155 row_write_positions[row]++; // 该行写入位置 +1
156 }
157}

159// 构造函数：CSR 格式(0-based)数据构造 CSR 稀疏矩阵
160SparseMatrixCSR::SparseMatrixCSR(const size_t rows, const size_t cols,
161 const vector<double>& values,
162 const vector<size_t>& col_indices,
163 const vector<size_t>& row_ptrs) :
164 rows(rows), cols(cols), values(values), col_indices(col_indices), row_ptrs(row_ptrs) {
165 // 检查输入参数
166 if (rows == 0) {
167 throw invalid_argument("错误：矩阵的行数必须大于 0。");
168 }
169 if (cols == 0) {
170 throw invalid_argument("错误：矩阵的列数必须大于 0。");
171 }
172 if (values.size() != col_indices.size()) {
173 throw invalid_argument("错误：CSR 格式错误，列指标、值的数量必须一致。");
174 }
175 if (row_ptrs.size() != rows + 1) {
176 throw invalid_argument("错误：CSR 格式错误，行指针的数量必须为 rows + 1。");
177 }
178 for (size_t i = 0; i < values.size(); ++i) {
179 if (col_indices[i] >= cols) {
180 throw out_of_range("错误：CSR 格式错误，列索引超出矩阵维度。");
181 }
182 }
183 for (size_t i = 0; i < rows; ++i) {
184 if (row_ptrs[i] > row_ptrs[i + 1]) {
185 throw invalid_argument("错误：CSR 格式错误，行指针必须递增。");
186 }
187 }
188 // 不能包含显式零元素
189 for (double val : values) {
190 if (val == 0.0) {
191 throw invalid_argument("错误：CSR 格式错误，不能包含显式零元素。");
192 }
193 }
194 // 行指针的第一个元素必须为 0
195 if (row_ptrs[0] != 0) {
196 throw invalid_argument("错误：CSR 格式错误，行指针的第一个元素必须为 0。");
197 }
198 // 行指针的最后一个元素必须等于 values.size()
199 if (row_ptrs[rows] != values.size()) {
200 throw invalid_argument("错误：CSR 格式错误，行指针的最后一个元素必须等于非零元素个数。");
201 }
202 // 检查输入必须无重复、良排序：每一行内的列索引必须严格递增
203 for (size_t i = 0; i < rows; ++i) {
204 const size_t start = row_ptrs[i];
205 const size_t end = row_ptrs[i + 1];
206 // 当前行没有非零元素，跳过
207 if (start == end) {
208 continue;
209 }
210 // 检查当前行的列索引是否严格递增
211 for (size_t j = start + 1; j < end; ++j) {
212 if (col_indices[j - 1] >= col_indices[j]) {
213 throw invalid_argument("错误：CSR 格式错误，每行内的列索引必须严格递增。");
214 }
215 }
216 }
217}

219// 构造函数：从文件(1-based)构造 CSR 稀疏矩阵
220SparseMatrixCSR::SparseMatrixCSR(const string& filename) {
221 // --- 1. 获取文件后缀名 ---
222 size_t dot_pos = filename.find_last_of('.');
223 if (dot_pos == string::npos || dot_pos == filename.length() - 1) {
224 throw runtime_error("错误：文件没有后缀名，请提供 .coo 或 .mtx 后缀以检测格式");
225 }
226
227 // 提取后缀（例如 ".coo"）
228 string ext = filename.substr(dot_pos); // 如 ".coo"
229 // 转小写（可选，使判断更鲁棒，如 .COO 也能识别）
230 for (char &c : ext) c = static_cast<char>(tolower(static_cast<unsigned char>(c)));
231
232 // --- 2. 根据后缀名选择处理逻辑 ---
233 if (ext == ".coo") {
234 // === COO 格式：每行是 row col value ===
235
236 ifstream file(filename);
237 if (!file.is_open()) {
238 throw runtime_error("无法打开 .coo 文件: " + filename);
239 }
240
241 vector<size_t> coo_rows;
242 vector<size_t> coo_cols;
243 vector<double> coo_vals;
244
245 string line;
246 size_t num_rows, num_cols, nnz;
247 bool got_first_line = false;
248 while (getline(file, line)) {
249 // 跳过空行和注释（以 # 或 % 开头）
250 if (line.empty() || line[0] == '#' || line[0] == '%') {
251 continue;
252 }
253
254 istringstream iss(line);
255 size_t row, col;
256 double val;
257
258 // 解析矩阵规模信息行
259 if (!got_first_line) {
260 got_first_line = true;
261 if (!(iss >> num_rows >> num_cols >> nnz)) {
262 throw runtime_error("文件格式错误，无法解析矩阵规模信息行: " + line);
263 }
264 continue;
265 }
266
267 // 解析 COO 行
268 if (!(iss >> row >> col >> val)) {
269 throw runtime_error("文件格式错误，无法解析 COO 行: " + line);
270 }
271
272 coo_rows.push_back(row);
273 coo_cols.push_back(col);
274 coo_vals.push_back(val);
275 }
276
277 // 调用已有的 COO -> CSR 构造函数
278 *this = SparseMatrixCSR(num_rows, num_cols, coo_rows, coo_cols, coo_vals);
279 if(getNNZ() != nnz) {
280 throw runtime_error("警告：文件中声明的非零元数量与实际读取的不符。");
281 }
282
283 } else if (ext == ".mtx") {
284 // === Matrix Market 格式：未来支持 ===
285 throw runtime_error("未来支持：Matrix Market 格式 (.mtx)，当前暂未实现");
286
287 } else {
288 // === 其他不支持的格式 ===
289 throw runtime_error("错误：不支持的文件格式 '" + ext + "'，请使用 .coo 或 .mtx 后缀");
290 }
291}

363// 设置CSR稀疏矩阵元素(1-based)
364void SparseMatrixCSR::setValue(vector<tuple<size_t, size_t, double>> elements) {
365 // 1. 根据行列索引对elements升序排列
366 sort(elements.begin(), elements.end(),
367 [](const tuple<size_t, size_t, double>& a, const tuple<size_t, size_t, double>& b) {
368 if (get<0>(a) != get<0>(b)) {
369 return get<0>(a) < get<0>(b); // 按行号升序
370 } else if (get<1>(a) != get<1>(b)) {
371 return get<1>(a) < get<1>(b); // 同一行，按列号升序
372 } else {
373 throw invalid_argument("新插入的元素包含位置完全相同的元素。");
374 }
375 });
376 // 2. 对每一行，插入列指标、值，更新行指针（同时检查输入合法性），然后排序
377 if (get<0>(elements[0]) == 0) {
378 throw invalid_argument("插入元素的坐标（行）请使用 1-based 记法，不应出现0。");
379 }
380 if (get<0>(elements[elements.size()-1]) > getRows()) {
381 throw invalid_argument("插入元素的坐标（行）超出矩阵维度。");
382 }
383 size_t elem_id = 0;
384 size_t insert_ptr = 0;
385 int update_row_ptrs = 0;
386 for (size_t row_id = 0; row_id < getRows(); ++row_id) {
387 // 更新行指针
388 row_ptrs[row_id+1] += update_row_ptrs;
389 if (elem_id == elements.size()) {
390 continue;
391 }
392 if (get<1>(elements[elem_id]) == 0) {
393 throw invalid_argument("插入元素的坐标（列）请使用 1-based 记法，不应出现0。");
394 }
395 if (get<1>(elements[elem_id]) > getCols()) {
396 throw invalid_argument("插入元素的坐标（列）超出矩阵维度。");
397 }
398 while (get<0>(elements[elem_id]) == row_id + 1) {
404 // 插入/更新/删除元素
405 insert_ptr = row_ptrs[row_id];
406 while (insert_ptr < row_ptrs[row_id+1] && col_indices[insert_ptr] < get<1>(elements[elem_id])-1) {
407 ++insert_ptr;
408 }
409 if (insert_ptr == row_ptrs[row_id+1] || col_indices[insert_ptr] > get<1>(elements[elem_id])-1) {
410 // 插入新元素
411 if (get<2>(elements[elem_id]) != 0.0) {
412 col_indices.insert(col_indices.begin() + insert_ptr, get<1>(elements[elem_id])-1);
413 values.insert(values.begin() + insert_ptr, get<2>(elements[elem_id]));
414 ++row_ptrs[row_id+1];
415 ++update_row_ptrs;
416 }
417 } else {
418 if (get<2>(elements[elem_id]) != 0.0) {
419 // 更新旧元素
420 values[insert_ptr] = get<2>(elements[elem_id]);
421 } else {
422 // 删除旧元素
423 col_indices.erase(col_indices.begin() + insert_ptr);
424 values.erase(values.begin() + insert_ptr);
425 --row_ptrs[row_id+1];
426 --update_row_ptrs;
427 }
428 }
429 ++elem_id;
430 if (elem_id == elements.size()) {
431 break;
432 }
433 }
434 }
435}

459double SparseMatrixCSR::operator()(const size_t row, const size_t col) const {
460 if(row == 0) {

530// 矩阵转置
531SparseMatrixCSR SparseMatrixCSR::transpose() const noexcept {
532 const size_t num_rows = this->cols; // 转置后行数 = 原列数
533 const size_t num_cols = this->rows; // 转置后列数 = 原行数
534
535 // 如果矩阵为空，直接返回空矩阵
536 if (this->values.empty()) {
537 return SparseMatrixCSR(num_rows, num_cols);
538 }
539
540 // Step 1: 统计原矩阵中每一列（即转置后每一行）有多少非零元素
541 vector<size_t> col_counts(num_rows, 0); // 大小为转置后行数 = 原列数
542 for (size_t k = 0; k < this->values.size(); ++k) {
543 size_t col = this->col_indices[k];
544 col_counts[col]++;
545 }
546
547 // Step 2: 计算转置后的 row_ptrs（即列的前缀和）
548 vector<size_t> transposed_row_ptrs(num_rows + 1, 0);
549 for (size_t i = 0; i < num_rows; ++i) {
550 transposed_row_ptrs[i + 1] = transposed_row_ptrs[i] + col_counts[i];
551 }
552
553 // Step 3: 填充转置后的 values, col_indices
554 vector<double> transposed_values(this->values.size());
555 vector<size_t> transposed_col_indices(this->col_indices.size());
556
557 // 辅助数组：记录每一列（转置行）当前写入位置
558 vector<size_t> current_pos(num_rows, 0);
559
560 for (size_t i = 0; i < this->rows; ++i) {
561 for (size_t k = this->row_ptrs[i]; k < this->row_ptrs[i + 1]; ++k) {
562 size_t col = this->col_indices[k]; // 原列号 → 转置后行号
563 size_t pos = transposed_row_ptrs[col] + current_pos[col];
564 transposed_values[pos] = this->values[k];
565 transposed_col_indices[pos] = i; // 原行号 → 转置后列号
566 current_pos[col]++;
567 }
568 }
569
570 // Step 4: 构造并返回新的 CSR（转置后矩阵）
571 return SparseMatrixCSR(num_rows, num_cols,
572 transposed_values,
573 transposed_col_indices,
574 transposed_row_ptrs);
575}

577// 矩阵范数
578double SparseMatrixCSR::norm(const string& norm_type) const {
579 if (norm_type == "row" || norm_type == "inf" || norm_type == "infinity" || norm_type == "INF" || norm_type == "INFINITY") {
580 // 行和范数（每行绝对值之和的最大值）→ ∞-范数
581 double max_row_sum = 0.0;
582 for (size_t i = 0; i < rows; ++i) {
583 double row_sum = 0.0;
584 for (size_t k = row_ptrs[i]; k < row_ptrs[i + 1]; ++k) {
585 row_sum += fabs(values[k]); // 累加绝对值
586 }
587 max_row_sum = max(max_row_sum, row_sum);
588 }
589 return max_row_sum;
590 } else if (norm_type == "col" || norm_type == "column" || norm_type == "1" || norm_type == "one" || norm_type == "ONE") {
591 // 列和范数（每列绝对值之和的最大值）→ 1-范数
592 vector<double> col_sums(cols, 0.0); // 每列的绝对值之和
593 for (size_t i = 0; i < rows; ++i) {
594 for (size_t k = row_ptrs[i]; k < row_ptrs[i + 1]; ++k) {
595 size_t col = col_indices[k];
596 col_sums[col] += fabs(values[k]);
597 }
598 }
599 double max_col_sum = 0.0;
600 for (double sum : col_sums) {
601 max_col_sum = max(max_col_sum, sum);
602 }
603 return max_col_sum;
604 } else if (norm_type == "fro" || norm_type == "Fro" || norm_type == "FRO" || norm_type == "frobenius" || norm_type == "Frobenius" || norm_type == "FROBENIUS") {
605 // Frobenius 范数：sqrt( sum(|A_ij|^2) )
606 double sum_sq = 0.0;
607 for (double val : values) {
608 sum_sq += val * val; // 已经是非负，不需要 fabs
609 }
610 return sqrt(sum_sq);
611 } else if (norm_type == "max" || norm_type == "maximum" || norm_type == "MAX" || norm_type == "MAXIMUM") {
612 // 按元素最大范数：max( |A_ij| )
613 double max_val = 0.0;
614 for (double val : values) {
615 max_val = max(max_val, fabs(val));
616 }
617 return max_val;
618 } else {
619 throw invalid_argument("不支持的范数类型：" + norm_type);
620 }
621}

792// 矩阵加法(+=)
793SparseMatrixCSR& SparseMatrixCSR::operator+=(const SparseMatrixCSR& other) {
794 // 检查维度是否匹配
795 if (this->rows != other.rows || this->cols != other.cols) {
796 throw invalid_argument("矩阵相加要求两矩阵大小相同");
797 }
798
799 // 临时存储结果
800 vector<double> result_values;
801 vector<size_t> result_col_indices;
802 vector<size_t> result_row_ptrs(this->rows + 1, 0);
803
804 // 逐行合并
805 for (size_t i = 0; i < this->rows; ++i) {
806 size_t pos1 = this->row_ptrs[i]; // 当前矩阵当前行的起始位置
807 size_t pos2 = other.row_ptrs[i]; // 另一个矩阵当前行的起始位置
808 size_t end1 = this->row_ptrs[i + 1]; // 当前矩阵当前行的结束位置
809 size_t end2 = other.row_ptrs[i + 1]; // 另一个矩阵当前行的结束位置
810
811 // 归并两个有序列表（按列索引）
812 while (pos1 < end1 && pos2 < end2) {
813 size_t col1 = this->col_indices[pos1];
814 size_t col2 = other.col_indices[pos2];
815
816 if (col1 < col2) {
817 // 只存在于当前矩阵
818 result_values.push_back(this->values[pos1]);
819 result_col_indices.push_back(col1);
820 pos1++;
821 } else if (col1 > col2) {
822 // 只存在于另一个矩阵
823 result_values.push_back(other.values[pos2]);
824 result_col_indices.push_back(col2);
825 pos2++;
826 } else {
827 // 两个矩阵都有，相加
828 double sum = this->values[pos1] + other.values[pos2];
829 if (abs(sum) > 1e-12) {
830 result_values.push_back(sum);
831 result_col_indices.push_back(col1);
832 }
833 pos1++;
834 pos2++;
835 }
836 }
837
838 // 处理剩余元素（当前矩阵）
839 while (pos1 < end1) {
840 result_values.push_back(this->values[pos1]);
841 result_col_indices.push_back(this->col_indices[pos1]);
842 pos1++;
843 }
844
845 // 处理剩余元素（另一个矩阵）
846 while (pos2 < end2) {
847 result_values.push_back(other.values[pos2]);
848 result_col_indices.push_back(other.col_indices[pos2]);
849 pos2++;
850 }
851
852 result_row_ptrs[i + 1] = result_values.size();
853 }
854
855 // 更新当前矩阵
856 this->values = move(result_values);
857 this->col_indices = move(result_col_indices);
858 this->row_ptrs = move(result_row_ptrs);
859
860 return *this;
861}
