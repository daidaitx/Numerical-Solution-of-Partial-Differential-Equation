40 // 获取网格点的数量
41 size_t total_points = grid_points.size();
42
43 // 初始化右端项 b
44 vector<double> b(total_points, 0.0);
45
46 // 初始化CSR稀疏矩阵 A 的三要素
47 vector<double> values(total_points * 5); // 预分配空间，最多每行5个非零元素
48 vector<size_t> col_indices(total_points * 5); // 预分配空间，最多每行5个非零元素
49 vector<size_t> row_ptrs(total_points + 1, 0); // row_ptrs 的大小固定为 total_points + 1
50
51 // 按x方向构造CSR稀疏矩阵 A 和右端项 b
52 for (size_t row_idx = 0; row_idx < total_points; ++row_idx) { // row_idx 既是矩阵行索引，也是当前中心网格点的拉直索引
53 /**********************************************************************************************
54 * 检查每一行，记录： *
55 * nnz_in_one_row 每行非零元素数 {1, 2, 3, 4, 5} *
56 * idx_*_* < total_points 是否存在四个内部邻近点 {True, False} *
57 * idx_*_* 四个内部邻近点拉直索引 {0, 1, ... , total_points-1, total_points} *
58 * l_*_* 到四个邻居的距离 [0, h] *
59 * g_*_* 边界条件 (-∞, +∞) *
60 * f_center 中心点的源项值 (-∞, +∞) *
61 ******************************************************************************************/
62
63 // 重置 nnz_in_one_row ： 既是矩阵A的当前行的非零元素数，也是当前中心网格点的邻近网格点数，包括中心网格点自己，该值不超过5
64 size_t nnz_in_one_row = 0;
65
66 // 检查-x邻居
67 size_t idx_minus_x = find(grid_indices.begin(), grid_indices.end(), make_pair(grid_indices[row_idx].first - 1, grid_indices[row_idx].second)) - grid_indices.begin(); // idx_minus_x 是 -x 邻居的拉直索引；如果不存在，则等于 total_points
68 bool has_minus_x = (idx_minus_x < total_points); // 是否存在-x内部邻近点
69 double l_minus_x = hx; // 到-x邻居的距离，默认为 hx，即规则网格距离
70 double g_minus_x = 0.0; // -x邻居的边界值，默认为 0.0，即不贡献边界值
71 if (has_minus_x) { // 存在-x内部邻近点
72 nnz_in_one_row++; // 本行非零元素数++
73 }
74 else { // 不存在-x内部邻近点，左侧靠近边界
75 // 计算到左侧边界邻交点的距离l_minus_x、左侧边界邻交点的值g_minus_x
76 tuple<double, double, pair<double, double>> boundary_info = irregularCalculator(polygon_vertices, grid_points[row_idx], "left", hx, g);
77 l_minus_x = get<0>(boundary_info);
78 g_minus_x = get<1>(boundary_info);
79 }
80
81 // 检查-y邻居
82 size_t idx_minus_y = find(grid_indices.begin(), grid_indices.end(), make_pair(grid_indices[row_idx].first, grid_indices[row_idx].second - 1)) - grid_indices.begin(); // idx_minus_y 是 -y 邻居的拉直索引；如果不存在，则等于 total_points
83 bool has_minus_y = (idx_minus_y < total_points); // 是否存在-y内部邻近点
84 double l_minus_y = hy; // 到-y邻居的距离，默认为 hy，即规则网格距离
85 double g_minus_y = 0.0; // -y邻居的边界值，默认为 0.0，即不贡献边界值
86 if (has_minus_y) { // 存在-y内部邻近点
87 nnz_in_one_row++; // 本行非零元素数++
88 }
89 else { // 不存在-y内部邻近点，底部靠近边界
90 // 计算到底部边界邻交点的距离l_minus_y、底部边界邻交点的值g_minus_y
91 tuple<double, double, pair<double, double>> boundary_info = irregularCalculator(polygon_vertices, grid_points[row_idx], "bottom", hy, g);
92 l_minus_y = get<0>(boundary_info);
93 g_minus_y = get<1>(boundary_info);
94 }
95
96 // 中心点
97 nnz_in_one_row++; // 一定有值，本行非零元素数++
98 double f_center = f(grid_points[row_idx].first, grid_points[row_idx].second); // 中心点的源项值
99
100 // 检查+y邻居
101 size_t idx_plus_y = find(grid_indices.begin(), grid_indices.end(), make_pair(grid_indices[row_idx].first, grid_indices[row_idx].second + 1)) - grid_indices.begin(); // idx_plus_y 是 +y 邻居的拉直索引；如果不存在，则等于 total_points
102 bool has_plus_y = (idx_plus_y < total_points); // 是否存在+y内部邻近点
103 double l_plus_y = hy; // 到+y邻居的距离，默认为 hy，即规则网格距离
104 double g_plus_y = 0.0; // +y邻居的边界值，默认为 0.0，即不贡献边界值
105 if (has_plus_y) { // 存在+y内部邻近点
106 nnz_in_one_row++; // 本行非零元素数++
107 }
108 else { // 不存在+y内部邻近点，顶部靠近边界
109 // 计算到顶部边界邻交点的距离l_plus_y、顶部边界邻交点的值g_plus_y
110 tuple<double, double, pair<double, double>> boundary_info = irregularCalculator(polygon_vertices, grid_points[row_idx], "top", hy, g);
111 l_plus_y = get<0>(boundary_info);
112 g_plus_y = get<1>(boundary_info);
113 }
114
115 // 检查+x邻居
116 size_t idx_plus_x = find(grid_indices.begin(), grid_indices.end(), make_pair(grid_indices[row_idx].first + 1, grid_indices[row_idx].second)) - grid_indices.begin(); // idx_plus_x 是 +x 邻居的拉直索引；如果不存在，则等于 total_points
117 bool has_plus_x = (idx_plus_x < total_points); // 是否存在+x内部邻近点
118 double l_plus_x = hx; // 到+x邻居的距离，默认为 hx，即规则网格距离
119 double g_plus_x = 0.0; // +x邻居的边界值，默认为 0.0，即不贡献边界值
120 if (has_plus_x) { // 存在+x内部邻近点
121 nnz_in_one_row++; // 本行非零元素数++
122 }
123 else { // 不存在+x内部邻近点，右侧靠近边界
124 // 计算到右侧边界邻交点的距离l_plus_x、右侧边界邻交点的值g_plus_x
125 tuple<double, double, pair<double, double>> boundary_info = irregularCalculator(polygon_vertices, grid_points[row_idx], "right", hx, g);
126 l_plus_x = get<0>(boundary_info);
127 g_plus_x = get<1>(boundary_info);
128 }
129
130 /*****************************************
131 * 更新 row_ptrs, col_indices, values, b *
132 ****************************************/
133
134 // 递归更新 row_ptrs
135 row_ptrs[row_idx + 1] = row_ptrs[row_idx] + nnz_in_one_row;
136
137 // 更新 col_indices, values, b
138 size_t elem_ptr = row_ptrs[row_idx];
139
140 // -x内部邻近点
141 if (has_minus_x) { // 存在-x内部邻近点
142 col_indices[elem_ptr] = idx_minus_x; // 列索引
143 values[elem_ptr] = - 2.0 / (l_minus_x * (l_minus_x + l_plus_x)); // 元素值
144 elem_ptr++;
145 }
146 else { // 不存在-x内部邻近点，左侧靠近边界
147 b[row_idx] += (2.0 * g_minus_x) / (l_minus_x * (l_minus_x + l_plus_x)); // 右端项贡献
148 }
149
150 // -y内部邻近点
151 if (has_minus_y) { // 存在-y内部邻近点
152 col_indices[elem_ptr] = idx_minus_y; // 列索引
153 values[elem_ptr] = - 2.0 / (l_minus_y * (l_minus_y + l_plus_y)); // 元素值
154 elem_ptr++;
155 }
156 else { // 不存在-y内部邻近点，底部靠近边界
157 b[row_idx] += (2.0 * g_minus_y) / (l_minus_y * (l_minus_y + l_plus_y)); // 右端项贡献
158 }
159
160 // 中心元素
161 col_indices[elem_ptr] = row_idx; // 列索引 = 行索引，即位于矩阵对角线位置
162 values[elem_ptr] = 2.0 / (l_minus_x * l_plus_x) + 2.0 / (l_minus_y * l_plus_y); // 元素值
163 elem_ptr++;
164 b[row_idx] += f_center; // 右端项贡献
165
166 // +y内部邻近点
167 if (has_plus_y) { // 存在+y内部邻近点
168 col_indices[elem_ptr] = idx_plus_y; // 列索引
169 values[elem_ptr] = - 2.0 / (l_plus_y * (l_minus_y + l_plus_y)); // 元素值
170 elem_ptr++;
171 }
172 else { // 不存在+y内部邻近点，顶部靠近边界
173 b[row_idx] += (2.0 * g_plus_y) / (l_plus_y * (l_minus_y + l_plus_y)); // 右端项贡献
174 }
175
176 // +x内部邻近点
177 if (has_plus_x) { // 存在+x内部邻近点
178 col_indices[elem_ptr] = idx_plus_x; // 列索引
179 values[elem_ptr] = - 2.0 / (l_plus_x * (l_minus_x + l_plus_x)); // 元素值
180 elem_ptr++;
181 }
182 else { // 不存在+x内部邻近点，右侧靠近边界
183 b[row_idx] += (2.0 * g_plus_x) / (l_plus_x * (l_minus_x + l_plus_x)); // 右端项贡献
184 }
185 }
186
187 // 调整values和col_indices的大小
188 size_t total_nnz = row_ptrs[total_points];
189 values.resize(total_nnz);
190 col_indices.resize(total_nnz);
191
192 // 构造和输出
193 SparseMatrixCSR A(total_points, total_points, values, col_indices, row_ptrs);
194 pair<SparseMatrixCSR, vector<double>> result = {A, b};
195 return result;
196 }
