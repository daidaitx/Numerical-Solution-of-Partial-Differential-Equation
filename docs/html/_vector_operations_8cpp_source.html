14double norm(const vector<double>& v, const string& normType) {
15 if (normType == "2") {
16 double sum = 0.0;
17 for (double val : v) {
18 sum += val * val;
19 }
20 return sqrt(sum);
21 } else if (normType == "1") {
22 double sum = 0.0;
23 for (double val : v) {
24 sum += abs(val);
25 }
26 return sum;
27 } else if (normType == "inf") {
28 if (v.empty()) return 0.0;
29 double max_val = abs(v[0]);
30 for (size_t i = 1; i < v.size(); ++i) {
31 max_val = max(max_val, abs(v[i]));
32 }
33 return max_val;
34 } else {
35 throw invalid_argument("不支持的范数类型: " + normType);
36 }
37}

169 ifstream fin(filename);
170 if (!fin) {
171 throw invalid_argument("无法打开文件: " + filename);
172 }
173
174 vector<vector<double>> result;
175 string line;
176
177 while (getline(fin, line)) { // 按行读取
178 if (line.empty()) {
179 continue; // 跳过空行
180 }
181
182 vector<double> row;
183 stringstream ss(line);
184 double value;
185
186 while (ss >> value) { // 从当前行中提取数字
187 row.push_back(value);
188 }
189
190 if (!row.empty()) {
191 result.push_back(row); // 将这一行的数字作为一个 vector 加入结果
192 }
193 }
194
195 fin.close();
196 return result;

317void print(const vector<vector<double>>& matrix, int precision) {
318 // 检查矩阵是否为空
319 if (matrix.empty()) {
320 cout << "矩阵大小: ( 0 x 0 )" << endl;
321 cout << "(空矩阵)" << endl;
322 return;
323 }
324
325 size_t rows = matrix.size();
326 size_t cols = matrix[0].size();
327
328 // 检查矩阵的合法性：所有行的列数是否相同
329 for (size_t i = 1; i < rows; ++i) {
330 if (matrix[i].size() != cols) {
331 cout << "错误：矩阵行大小不一致！" << endl;
332 return;
333 }
334 }
335
336 // 处理0列的情况
337 if (cols == 0) {
338 cout << "矩阵大小: ( " << rows << " x 0 )" << endl;
339 cout << "(空矩阵)" << endl;
340 return;
341 }
342
343 cout << "矩阵大小: ( " << rows << " x " << cols << " )" << endl;
344
345 size_t max_width = 0;
346
347 // 遍历所有元素，计算它们格式化后的最大显示宽度
348 for (const auto& row : matrix) {
349 for (double val : row) {
350 ostringstream oss;
351 oss << fixed << setprecision(precision);
352 oss << val;
353 string s = oss.str();
354 if (s.length() > max_width) {
355 max_width = s.length();
356 }
357 }
358 }
359
360 // 打印矩阵内容
361 for (const auto& row : matrix) {
362 for (double val : row) {
363 ostringstream os;
364 os << fixed << setprecision(precision);
365 os << val;
366 cout << setw(max_width) << os.str() << " ";
367 }
