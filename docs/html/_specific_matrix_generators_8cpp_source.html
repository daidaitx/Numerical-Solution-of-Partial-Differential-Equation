17vector<double> flattenMatrix(const vector<vector<double>>& matrix_2d) {
18 if (matrix_2d.empty()) return {};
19
20 size_t M = matrix_2d.size();
21 size_t N = matrix_2d[0].size();
22 vector<double> flattened(M * N);
23
24 for (size_t i = 0; i < M; ++i) {
25 if (matrix_2d[i].size() != N) {
26 throw invalid_argument("Input matrix has inconsistent row sizes");
27 }
28 for (size_t j = 0; j < N; ++j) {
29 flattened[i * N + j] = matrix_2d[i][j];
30 }
31 }
32 return flattened;
33}

56SparseMatrixCSR generatePoissonMatrix2D(const PoissonConfig2D& config) {
57 size_t M = config.M - 1; // 网格数减1为实际内部点数
58 size_t N = config.N - 1;
59 size_t total_points = M * N; // 总未知格点数
60
61 double hx = config.hx();
62 double hy = config.hy();
63 double hx2 = hx * hx;
64 double hy2 = hy * hy;
65
66 // 五点差分格式的系数(-Δ)
67 double center_coeff = 2.0/hx2 + 2.0/hy2;
68 double x_coeff = -1.0/hx2;
69 double y_coeff = -1.0/hy2;
70
71 vector<double> values;
72 vector<size_t> col_indices;
73 vector<size_t> row_ptrs(total_points + 1, 0);
74
75 // 按行拉直：预先计算每行的非零元素个数
76 for (size_t i = 0; i < M; ++i) { // i = 0:M-1
77 for (size_t j = 0; j < N; ++j) { // j = 0:N-1
78 size_t row_idx = index2Dto1D(i, j, N);
79 size_t nnz_in_row = 0;
80
81 // 检查-x邻居
82 if (i > 0) nnz_in_row++;
83 // 检查-y邻居
84 if (j > 0) nnz_in_row++;
85 // 中心点
86 nnz_in_row++;
87 // 检查+y邻居
88 if (j < N - 1) nnz_in_row++;
89 // 检查+x邻居
90 if (i < M - 1) nnz_in_row++;
91
92 row_ptrs[row_idx + 1] = row_ptrs[row_idx] + nnz_in_row;
93 }
94 }
95
96 // 调整values和col_indices的大小
97 size_t total_nnz = row_ptrs[total_points];
98 values.resize(total_nnz);
99 col_indices.resize(total_nnz);
100
101 // 填充矩阵元素
102 for (size_t i = 0; i < M; ++i) { // i = 0:M-1
103 for (size_t j = 0; j < N; ++j) { // j = 0:N-1
104 size_t row_idx = index2Dto1D(i, j, N);
105 size_t elem_ptr = row_ptrs[row_idx];
106
108 // -x邻居 (i-1, j)
109 if (i > 0) {
110 size_t bottom_idx = index2Dto1D(i-1, j, N);
111 values[elem_ptr] = y_coeff;
112 col_indices[elem_ptr] = bottom_idx;
113 elem_ptr++;
114 }
115 // -y邻居 (i, j-1)
116 if (j > 0) {
117 size_t left_idx = index2Dto1D(i, j-1, N);
118 values[elem_ptr] = x_coeff;
119 col_indices[elem_ptr] = left_idx;
120 elem_ptr++;
121 }
122 // 中心元素
123 values[elem_ptr] = center_coeff;
124 col_indices[elem_ptr] = row_idx;
125 elem_ptr++;
126 // +y邻居 (i, j+1)
127 if (j < N - 1) {
128 size_t right_idx = index2Dto1D(i, j+1, N);
129 values[elem_ptr] = x_coeff;
130 col_indices[elem_ptr] = right_idx;
131 elem_ptr++;
132 }
133 // +x邻居 (i+1, j)
134 if (i < M - 1) {
135 size_t top_idx = index2Dto1D(i+1, j, N);
136 values[elem_ptr] = y_coeff;
137 col_indices[elem_ptr] = top_idx;
138 elem_ptr++;
139 }
140 }
141 }
142
143 return SparseMatrixCSR(total_points, total_points, values, col_indices, row_ptrs);
144}

148vector<double> generatePoissonRHS2D(
149 const PoissonConfig2D& config,
150 function<double(double, double)> f,
151 function<double(double, double)> g) {
152
153 size_t M = config.M - 1;
154 size_t N = config.N - 1;
155 double hx = config.hx();
156 double hy = config.hy();
157 double hx2 = hx * hx;
158 double hy2 = hy * hy;
159
160 vector<double> rhs(M * N, 0.0);
161
162 // 计算网格点坐标并填充右端项
163 for (size_t i = 0; i < M; ++i) { // i = 0:M-1
164 double x = config.x_min + (i + 1) * hx; // 网格中心
165 for (size_t j = 0; j < N; ++j) { // j = 0:N-1
166 double y = config.y_min + (j + 1) * hy; // 网格中心
167 size_t idx = index2Dto1D(i, j, N);
168
169 // -x邻居 (i-1, j)
170 if (i == 0) {
171 rhs[idx] += g(x - hx, y) / hx2;
172 }
173 // -y邻居 (i, j-1)
174 if (j == 0) {
175 rhs[idx] += g(x, y - hy) / hy2;
176 }
177 // 中心元素：右端项为 f(x,y)
178 rhs[idx] += f(x, y);
179 // +y邻居 (i, j+1)
180 if (j == N - 1) {
181 rhs[idx] += g(x, y + hy) / hy2;
182 }
183 // +x邻居 (i+1, j)
184 if (i == M - 1) {
185 rhs[idx] += g(x + hx, y) / hx2;
186 }
187 }
188 }
189
190 return rhs;
191}

193vector<double> generatePoissonRHS2D(
194 const PoissonConfig2D& config,
195 const vector<vector<double>>& f_discrete,
196 const vector<vector<double>>& g_discrete) {
197
198 size_t M = config.M;
199 size_t N = config.N;
200
201 // 验证输入尺寸
202 if (f_discrete.size() != M || f_discrete[0].size() != N) {
203 throw invalid_argument("f_discrete dimensions do not match config M, N");
204 }
205 if (g_discrete.size() != M || g_discrete[0].size() != N) {
206 throw invalid_argument("g_discrete dimensions do not match config M, N");
207 }
208
209 vector<double> rhs(M * N, 0.0);
210
211 for (size_t i = 0; i < M; ++i) {
212 for (size_t j = 0; j < N; ++j) {
213 size_t idx = index2Dto1D(i, j, N);
214
215 // 内部点使用f_discrete
216 rhs[idx] = f_discrete[i][j];
217
218 // 边界点使用g_discrete
219 bool is_boundary = (i == 0) || (i == M - 1) || (j == 0) || (j == N - 1);
220 if (is_boundary) {
221 rhs[idx] = g_discrete[i][j];
222 }
223 }
224 }
225
226 return rhs;
227}

231PoissonProblem2D generatePoisson2D(
232 const PoissonConfig2D& config,
233 function<double(double, double)> f,
234 function<double(double, double)> g) {
235
236 PoissonProblem2D problem;
237 problem.config = config;
238
239 // 生成矩阵和右端项
240 problem.A = generatePoissonMatrix2D(config);
241 problem.b = generatePoissonRHS2D(config, f, g);
242
243 // 求解线性方程组
244 problem.solution_1d = problem.A.solve(problem.b);
245 problem.solution_2d = unflattenVector(problem.solution_1d, config.M - 1, config.N - 1);
246
247 return problem;
248}

267void PoissonProblem2D::saveToFiles(const string& basename) const {

276
277 // 保存网格
278 vector<vector<double>> X(config.M - 1, vector<double>(config.N - 1));
279 vector<vector<double>> Y(config.M - 1, vector<double>(config.N - 1));

281 for (size_t j = 0; j < config.N - 1; ++j) {
282 X[i][j] = config.x_min + (i + 1) * config.hx();
283 Y[i][j] = config.y_min + (j + 1) * config.hy();
284 }
285 }
286 VectorOps::writeMat(X, basename + "_X.mat");
287 VectorOps::writeMat(Y, basename + "_Y.mat");
