27vector<pair<double, double>> ComputePolygonBoundaryNormals(const vector<pair<double, double>>& polygon_vertices) {
28 vector<pair<double, double>> normals;
29 for (size_t i = 0; i < polygon_vertices.size(); ++i) {
30 size_t j = (i + 1) % polygon_vertices.size();
31 double dx = polygon_vertices[j].first - polygon_vertices[i].first;
32 double dy = polygon_vertices[j].second - polygon_vertices[i].second;
33 double normal_x = dy;
34 double normal_y = -dx;
35 double normal_norm = sqrt(normal_x * normal_x + normal_y * normal_y);
36 normals.push_back({normal_x / normal_norm, normal_y / normal_norm});
37 }
38 return normals;
39}

50bool isAtSegment(double x, double y, const pair<double, double>& A, const pair<double, double>& B) {
51 double dx = B.first - A.first;
52 double dy = B.second - A.second;
53 double len2 = dx * dx + dy * dy;
54 double t = ((x - A.first) * dx + (y - A.second) * dy) / len2;
55 if (t < 0.0 - 1e-10 || t > 1.0 + 1e-10) { // 点不在线段上
56 return false;
57 }
58 double x_proj = A.first + t * dx;
59 double y_proj = A.second + t * dy;
60 double dist2 = (x - x_proj) * (x - x_proj) + (y - y_proj) * (y - y_proj);
61 if (dist2 > 1e-10) { // 点不在线段上
62 return false;
63 }
64 return true;
65}

354 values[elem_ptr_plus_y] += scale * -2.0 * nu_y_minus_x / (l_plus_y * nu_l_l);
355 values[elem_ptr_center] -= scale * 2.0 * (nu_x_minus_x * l_plus_y - nu_y_minus_x * l_minus_x) / (l_plus_y * l_minus_x * nu_l_l);
356 }
357 else { // 四个内部邻近点都不存在，抛出异常
358 throw invalid_argument("网格剖分过于稀疏！");
359 }
360 }
361
362 if (extra_minus_y != 0.0) {
363 double nu_l_l = nu_y_minus_y * (l_minus_y + l_plus_y);
364 double coeff = -2.0 / (l_minus_y * (l_minus_y + l_plus_y));
365 double scale = extra_minus_y / coeff;
366 if (has_minus_x) { // -x内部邻近点存在
367 b[row_idx] -= scale * 2.0 * g_minus_y / nu_l_l;
368 values[elem_ptr_minus_x] += scale * 2.0 * nu_x_minus_y / (l_minus_x * nu_l_l);
369 values[elem_ptr_center] -= scale * 2.0 * (nu_y_minus_y * l_minus_x + nu_x_minus_y * l_minus_y) / (l_minus_x * l_minus_y * nu_l_l);
370 }
371 else if (has_plus_x) { // -x内部邻近点不存在，+x内部邻近点存在
372 b[row_idx] -= scale * 2.0 * g_minus_y / nu_l_l;
373 values[elem_ptr_plus_x] += scale * -2.0 * nu_x_minus_y / (l_plus_x * nu_l_l);
374 values[elem_ptr_center] -= scale * 2.0 * (nu_y_minus_y * l_plus_x - nu_x_minus_y * l_minus_y) / (l_plus_x * l_minus_y * nu_l_l);
375 }
376 else { // 四个内部邻近点都不存在，抛出异常
377 throw invalid_argument("网格剖分过于稀疏！");
378 }
379 }
380 }
381
382 // 调整values和col_indices的大小
383 size_t total_nnz = row_ptrs[total_points];
384 values.resize(total_nnz);
385 col_indices.resize(total_nnz);
386
387 // 构造和输出
388 SparseMatrixCSR A(total_points, total_points, values, col_indices, row_ptrs);
389 pair<SparseMatrixCSR, vector<double>> result = {A, b};
390 return result;
391 }
